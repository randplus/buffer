## Usage
```js
const random = require('@randplus/buffer');

console.log(random()) // '<Buffer >'
console.log(random(16)) // '<Buffer 26 bc 62 84 b0 94 5e c5 1a d5 fd 69 6f d3 78 b1>'
```
